@app.route('/miac/obter-dados', methods=['POST'])
def obter_dados():
    if 'username' not in session:
        return jsonify({'error': 'Usuário não autenticado'}), 403

    try:
        files = request.files.getlist('pdf_file')
        if not files:
            return jsonify({'error': 'Nenhum arquivo enviado.'}), 400

        resultados = []

        for file in files:
            if file.filename == '' or not file.filename.lower().endswith('.pdf'):
                continue  # Pula arquivos inválidos

            file_name = f"{int(time.time())}_{secure_filename(file.filename)}"
            file_path = os.path.join(
                UPLOAD_FOLDER2, file_name).replace("\\", "/")
            file.save(file_path)

            pdf_text = read_last_page(file_path)
            if pdf_text is None:
                resultados.append(
                    {'error': f'Erro ao ler o PDF: {file.filename}'})
                continue

            prompt = (
                f"Por favor, extraia as seguintes informações do texto literais MANDE SEM ASTERISICO, SO O TEXTO BRUTO, TIPO PROCEDIMENTO OPERACIONAL PADRAO, FLUXO E ETC, se voce não achar nao use o mais parecido, use apenas não localizado:\n"
                f"1. Data de Elaboração\n"
                f"2. Vencimento\n"
                f"3. Elaboradores\n"
                f"4. Número SEI que é tipo assim 23533.037366/2024-05\n"
                f"5. Organograma (extrair do código genérico no formato 'POP.UAP-CHUFC.006', onde 'UAP' é o Organograma, mande so a sigla)\n"
                f"6. Tipo de Documento (extrair do texto, procurando por 'Tipo do Documento' mande so o tipo)\n"
                f"7. Abrangência (extrair do código genérico no formato 'POP.UAP-CHUFC.006', onde 'CHUFC' é a Abrangência, SE APArecer CHUFC me retorne HUWC mande so a sigla)\n"
                f"8. Título do Documento\n"
                f"Texto:\n\n{pdf_text}"
            )
            gpt_response = send_to_gpt(prompt, api_key)
            if gpt_response is None:
                resultados.append(
                    {'error': f'Erro ao comunicar com o GPT para o arquivo: {file.filename}'})
                continue

            data_elaboracao = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Data de Elaboração" in line]
            vencimento = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Vencimento" in line]
            numero_sei = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Número SEI" in line]
            elaboradores = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Elaboradores" in line][0].split(", ")
            organograma = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Organograma" in line]
            tipo_documento = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Tipo de Documento" in line]
            abrangencia = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Abrangência" in line]
            titulo = [line.split(":")[1].strip() for line in gpt_response.split(
                "\n") if "Título" in line]

            resultados.append({
                'gpt_response': {
                    'data_elaboracao': data_elaboracao[0] if data_elaboracao else "Não localizado",
                    'vencimento': vencimento[0] if vencimento else "Não localizado",
                    'numero_sei': numero_sei[0] if numero_sei else "Não localizado",
                    'titulo': file.filename,
                    'organograma': organograma[0] if organograma else "Não localizado",
                    'tipo_documento': tipo_documento[0] if tipo_documento else "Não localizado",
                    # Novo campo
                    'abrangencia': abrangencia[0] if abrangencia else "Não localizado",
                    'titulo': titulo[0] if titulo else "Não localizado"
                },
                'elaboradores': elaboradores
            })

            os.remove(file_path)
